// ───────────────────────────────────────────────────────────── // LINGO — Google Apps Script (fixed streak/log) // ───────────────────────────────────────────────────────────── const WORDS_SHEET = 'Palabras'; const SENTENCES_SHEET = 'Frases'; const LOG_SHEET = 'Racha'; function doGet(e) { return handleRequest(e); } function doPost(e) { return handleRequest(e); } function handleRequest(e) { try { const params = e.parameter || {}; let body = {}; if (e.postData) { const dataParam = params.data; if (dataParam) { try { body = JSON.parse(dataParam); } catch(err) {} } else { try { body = JSON.parse(e.postData.contents || '{}'); } catch(err) {} } } const action = params.action || body.action || 'get'; const ss = SpreadsheetApp.getActiveSpreadsheet(); if (action === 'get') { return out({ ok: true, words: getWords(ss), sentences: getSentences(ss), practiceLog: getPracticeLog(ss) }); } if (action === 'saveAll') { if (body.words !== undefined) saveWords(ss, body.words); if (body.sentences !== undefined) saveSentences(ss, body.sentences); if (body.practiceLog !== undefined) mergePracticeLog(ss, body.practiceLog); return out({ ok: true }); } if (action === 'mergePracticeLog') { mergePracticeLog(ss, body.practiceLog); return out({ ok: true, practiceLog: getPracticeLog(ss) }); } // Legacy if (action === 'savePracticeLog') { mergePracticeLog(ss, body.practiceLog); return out({ ok: true }); } return out({ ok: false, error: 'Unknown action' }); } catch(err) { return out({ ok: false, error: err.message }); } } function out(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); } // ── WORDS function getWords(ss) { const sheet = ss.getSheetByName(WORDS_SHEET); if (!sheet) return []; const data = sheet.getDataRange().getValues(); if (data.length <= 1) return []; return data.slice(1).map(r => ({ id: r[0], en: r[1], es: r[2], pron: r[3], type: r[4], level: r[5] || '', example: r[6], status: r[7] || 'learning', lastPracticed: r[8] || null, practiceCount: Number(r[9]) || 0 })).filter(w => w.en); } function saveWords(ss, words) { let sheet = ss.getSheetByName(WORDS_SHEET); if (!sheet) sheet = ss.insertSheet(WORDS_SHEET); sheet.clearContents(); const header = [['id','en','es','pron','type','level','example','status','lastPracticed','practiceCount']]; sheet.getRange(1,1,1,10).setValues(header) .setFontWeight('bold') .setBackground('#2a1140') .setFontColor('#c169d6'); if (!words.length) return; const rows = words.map(w => [ w.id||'', w.en||'', w.es||'', w.pron||'', w.type||'', w.level||'', w.example||'', w.status||'learning', w.lastPracticed||'', w.practiceCount||0 ]); sheet.getRange(2, 1, rows.length, 10).setValues(rows); sheet.autoResizeColumns(1, 10); } // ── SENTENCES function getSentences(ss) { const sheet = ss.getSheetByName(SENTENCES_SHEET); if (!sheet) return []; const data = sheet.getDataRange().getValues(); if (data.length <= 1) return []; return data.slice(1).map(r => ({ id: r[0], es: r[1], status: r[2] || 'learning', lastPracticed: r[3] || null, translations: [r[4],r[5],r[6],r[7],r[8]].map(v => String(v||'').trim()).filter(Boolean) })).filter(s => s.es); } function saveSentences(ss, sentences) { let sheet = ss.getSheetByName(SENTENCES_SHEET); if (!sheet) sheet = ss.insertSheet(SENTENCES_SHEET); sheet.clearContents(); sheet.getRange(1,1,1,9) .setValues([['id','es','status','lastPracticed','en1','en2','en3','en4','en5']]) .setFontWeight('bold') .setBackground('#2a1140') .setFontColor('#c169d6'); if (!sentences.length) return; const rows = sentences.map(s => { const t = s.translations || []; return [s.id||'', s.es||'', s.status||'learning', s.lastPracticed||'', t[0]||'', t[1]||'', t[2]||'', t[3]||'', t[4]||'']; }); sheet.getRange(2, 1, rows.length, 9).setValues(rows); sheet.autoResizeColumns(1, 9); } // ── PRACTICE LOG ────────────────────────────────────────────── // // Sheet "Racha" layout: // A1: fecha (header) // A2..AN: YYYY-MM-DD dates, one per row, sorted, no duplicates // // mergePracticeLog: // 1. Reads ALL existing dates from the sheet into a Set // 2. Adds incoming dates from the client // 3. Writes the union back (sorted, deduplicated) // // NEVER uses clearContents() before reading — the read happens first, // then clearContents(), then the full merged set is written. // This is the key fix: the sheet always accumulates every date it has ever seen. function getPracticeLog(ss) { const sheet = ss.getSheetByName(LOG_SHEET); if (!sheet) return []; const lastRow = sheet.getLastRow(); if (lastRow <= 1) return []; const data = sheet.getRange(2, 1, lastRow - 1, 1).getValues(); return data .map(r => String(r[0] || '').trim()) .filter(d => /^\d{4}-\d{2}-\d{2}$/.test(d)); } function mergePracticeLog(ss, incomingDates) { let sheet = ss.getSheetByName(LOG_SHEET); if (!sheet) sheet = ss.insertSheet(LOG_SHEET); // 1. Read what is ALREADY in the sheet BEFORE touching anything const existing = new Set(getPracticeLog(ss)); // 2. Add incoming dates from the client if (incomingDates && Array.isArray(incomingDates)) { incomingDates.forEach(d => { const s = String(d || '').trim(); if (/^\d{4}-\d{2}-\d{2}$/.test(s)) existing.add(s); }); } // 3. NOW clear and rewrite the full merged, sorted set sheet.clearContents(); sheet.getRange(1, 1).setValue('fecha'); sheet.getRange(1, 1).setFontWeight('bold') .setBackground('#2a1140') .setFontColor('#c169d6'); if (existing.size === 0) return; const sorted = [...existing].sort(); // lexicographic sort works for YYYY-MM-DD const rows = sorted.map(d => [d]); sheet.getRange(2, 1, rows.length, 1).setValues(rows); }
